/*********************************************************************** * English Land LMS - Backend (Google Apps Script) * يربط منصة الطلاب/المحاضرين بجوجل شيت المكوّن من 7 شيتات: * Students, Teachers, Schedule, Materials, Attendance, Grades, Exams ***********************************************************************/ const SS = SpreadsheetApp.getActiveSpreadsheet(); const SHEETS = { STUDENTS: 'Students', TEACHERS: 'Teachers', SCHEDULE: 'Schedule', MATERIALS: 'Materials', ATTENDANCE: 'Attendance', GRADES: 'Grades', EXAMS: 'Exams' }; /* ==================== نقاط الدخول (Entry Points) ==================== */ function doGet(e) { try { const action = e.parameter.action; if (action === 'login') { return jsonOut(handleLogin(e.parameter.id, e.parameter.pass)); } return jsonOut({ error: 'إجراء (action) غير معروف' }); } catch (err) { return jsonOut({ error: 'خطأ في السيرفر: ' + err.message }); } } function doPost(e) { try { const p = e.parameter; const action = p.action; let result; switch (action) { case 'submitHomework': result = handleSubmitHomework(p); break; case 'submitGameScore': result = handleSubmitGameScore(p); break; case 'saveSchedule': result = handleSaveSchedule(p); break; case 'uploadExam': result = handleUploadExam(p); break; case 'gradeHomework': result = handleGradeHomework(p); break; case 'updateAttendance': result = handleUpdateAttendance(p); break; case 'updateStudentFullInfo': result = handleUpdateStudentFullInfo(p); break; default: result = { error: 'إجراء (action) غير معروف' }; } return jsonOut(result); } catch (err) { return jsonOut({ error: 'خطأ في السيرفر: ' + err.message }); } } function jsonOut(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } /* ==================== أدوات عامة لقراءة/كتابة الشيتات ==================== */ function getSheetData(sheetName) { const sheet = SS.getSheetByName(sheetName); if (!sheet) throw new Error('الشيت غير موجود: ' + sheetName); const values = sheet.getDataRange().getValues(); const headers = values[0]; const rows = []; for (let i = 1; i < values.length; i++) { const rowArr = values[i]; const isEmpty = rowArr.every(c => c === '' || c === null || c === undefined); if (isEmpty) continue; const obj = { __row: i + 1 }; headers.forEach((h, idx) => { if (h) obj[h] = rowArr[idx]; }); rows.push(obj); } return { sheet, headers, rows }; } function setCell(sheet, headers, rowNum, headerName, value) { const colIdx = headers.indexOf(headerName) + 1; if (colIdx > 0) sheet.getRange(rowNum, colIdx).setValue(value); } function parseJsonSafe(val) { if (!val) return {}; try { const parsed = JSON.parse(val); return (parsed && typeof parsed === 'object') ? parsed : {}; } catch (e) { return {}; } } function normId(v) { return String(v == null ? '' : v).trim().toUpperCase(); } /* ==================== تسجيل الدخول ==================== */ function handleLogin(id, pass) { if (!id || !pass) return { error: 'برجاء إدخال الـ ID وكلمة المرور' }; id = String(id).trim(); pass = String(pass).trim(); const { rows: teacherRows } = getSheetData(SHEETS.TEACHERS); const teacher = teacherRows.find(t => normId(t['ID']) === normId(id)); if (teacher) { if (String(teacher['كلمة المرور'] || '').trim() !== pass) { return { error: 'كلمة المرور غير صحيحة' }; } return buildTeacherResponse(teacher); } const { rows: studentRows } = getSheetData(SHEETS.STUDENTS); const student = studentRows.find(s => normId(s['Student ID']) === normId(id)); if (!student) return { error: 'الـ ID غير موجود، تأكد من كتابته بشكل صحيح' }; if (String(student['كلمة المرور'] || '').trim() !== pass) { return { error: 'كلمة المرور غير صحيحة' }; } return buildStudentResponse(student); } /* ==================== بناء استجابة الطالب ==================== */ function buildStudentResponse(student) { const sid = String(student['Student ID']).trim(); const course = String(student['Course'] || '').trim(); const group = (student['Group'] !== undefined && student['Group'] !== null) ? String(student['Group']).trim() : ''; const { rows: attRows } = getSheetData(SHEETS.ATTENDANCE); const attRow = attRows.find(a => normId(a['ID']) === normId(sid)); const attCounts = countAttendance(attRow); const attendanceDetails = {}; for (let i = 1; i <= 8; i++) { attendanceDetails['L' + i] = attRow ? (attRow['L' + i] || '') : ''; } const { rows: gradeRows } = getSheetData(SHEETS.GRADES); const gradeRow = gradeRows.find(g => normId(g['ID']) === normId(sid)) || {}; const { rows: materialRows } = getSheetData(SHEETS.MATERIALS); const materials = materialRows .filter(m => String(m['Course'] || '').trim().toLowerCase() === course.toLowerCase()) .map(m => ({ title: m['Title'] || '', link: m['Link'] || '', gameLink: m['Game Link'] || '' })); const exams = buildStudentExamsMap(course); return { role: 'student', student: { id: sid, name: String(student['Name'] || ''), course: course, group: group, schedule: String(student['Schedule'] || ''), teacher: String(student['Teacher'] || ''), feedback: String(student['Feedback'] || ''), payment: String(student['Payment'] || ''), percentage: formatPercentage(student['Percentage']), homework: { writing: student['Writing'] || '', shadowing: student['Shadowing'] || '', hwGrade: student['HW_Grade'] || '', hwFeedback: student['HW_Feedback'] || '' } }, attendance: attCounts, attendanceDetails: attendanceDetails, grades: { quiz1: gradeRow['Quiz1'] !== undefined ? gradeRow['Quiz1'] : '', quiz2: gradeRow['Quiz2'] !== undefined ? gradeRow['Quiz2'] : '', final: gradeRow['Final'] !== undefined ? gradeRow['Final'] : '' }, materials: materials, exams: exams, schedule: buildScheduleMap() }; } function formatPercentage(val) { if (val === undefined || val === null || val === '') return ''; if (typeof val === 'number') return Math.round(val * 100) + '%'; return String(val); } function countAttendance(attRow) { let total = 0, attended = 0; if (attRow) { for (let i = 1; i <= 8; i++) { const val = attRow['L' + i]; if (val === 'ح' || val === 'حاضر' || val === 'غ' || val === 'غائب') { total++; if (val === 'ح' || val === 'حاضر') attended++; } } } return { total: total, attended: attended, absences: total - attended }; } function buildStudentExamsMap(course) { const { rows } = getSheetData(SHEETS.EXAMS); const examMap = {}; const prefix = course + '_'; rows.forEach(r => { const key = String(r['Examkey'] || '').trim(); if (!key) return; if (key.toLowerCase().indexOf(prefix.toLowerCase()) === 0) { const typePart = key.substring(prefix.length); const normalizedKey = typePart.toLowerCase(); examMap[normalizedKey] = r['Examdata']; } }); return examMap; } /* ==================== جدول المواعيد ==================== */ function getScheduleRows() { const sheet = SS.getSheetByName(SHEETS.SCHEDULE); const values = sheet.getDataRange().getValues(); const rows = []; for (let i = 1; i < values.length; i++) { const [group, lecture, dateTime, zoom] = values[i]; if (!group && !lecture) continue; rows.push({ row: i + 1, group: group, lecture: lecture, dateTime: dateTime, zoom: zoom }); } return rows; } function buildScheduleMap() { const rows = getScheduleRows(); const map = {}; rows.forEach(r => { if (!r.group || !r.lecture) return; const key = String(r.group).trim() + '_' + String(r.lecture).trim(); let dateStr = ''; if (r.dateTime instanceof Date) { dateStr = Utilities.formatDate(r.dateTime, Session.getScriptTimeZone(), "yyyy-MM-dd'T'HH:mm:ss"); } else if (r.dateTime) { dateStr = String(r.dateTime); } map[key] = { date: dateStr, zoom: r.zoom ? String(r.zoom) : '' }; }); return map; } /* ==================== بناء استجابة المحاضر ==================== */ function getTeacherRole(teacher) { if (teacher['Role']) { const r = String(teacher['Role']).trim().toLowerCase(); return (r.indexOf('admin') > -1) ? 'superadmin' : 'teacher'; } return normId(teacher['ID']) === 'T101' ? 'superadmin' : 'teacher'; } function buildTeacherResponse(teacher) { const teacherName = String(teacher['Name'] || '').trim(); const role = getTeacherRole(teacher); const { rows: studentRows } = getSheetData(SHEETS.STUDENTS); const { rows: attRows } = getSheetData(SHEETS.ATTENDANCE); const { rows: gradeRows } = getSheetData(SHEETS.GRADES); const attMap = {}; attRows.forEach(a => { attMap[normId(a['ID'])] = a; }); const gradeMap = {}; gradeRows.forEach(g => { gradeMap[normId(g['ID'])] = g; }); let filteredStudents = studentRows; if (role !== 'superadmin') { filteredStudents = studentRows.filter(s => String(s['Teacher'] || '').trim() === teacherName); } const students = filteredStudents.map(s => buildStudentSummary(s, attMap, gradeMap)); const result = { role: role, name: teacherName, students: students, schedule: buildScheduleMap() }; if (role === 'superadmin') { const groupsSet = new Set(); studentRows.forEach(s => { if (s['Group'] !== undefined && s['Group'] !== null && s['Group'] !== '') groupsSet.add(String(s['Group']).trim()); }); result.groups = Array.from(groupsSet); } return result; } function buildStudentSummary(s, attMap, gradeMap) { const sid = String(s['Student ID']).trim(); const attRow = attMap[normId(sid)]; const attCounts = countAttendance(attRow); const gradeRow = gradeMap[normId(sid)] || {}; return { id: sid, name: String(s['Name'] || ''), group: (s['Group'] !== undefined && s['Group'] !== null) ? String(s['Group']) : '', course: String(s['Course'] || ''), payment: String(s['Payment'] || ''), feedback: String(s['Feedback'] || ''), homework: { writing: s['Writing'] || '', shadowing: s['Shadowing'] || '', hwGrade: s['HW_Grade'] || '', hwFeedback: s['HW_Feedback'] || '' }, attendance: attCounts, grades: { quiz1: gradeRow['Quiz1'] !== undefined ? gradeRow['Quiz1'] : '', quiz2: gradeRow['Quiz2'] !== undefined ? gradeRow['Quiz2'] : '', final: gradeRow['Final'] !== undefined ? gradeRow['Final'] : '' } }; } function handleSubmitHomework(p) { const { sheet, headers, rows } = getSheetData(SHEETS.STUDENTS); const row = rows.find(r => normId(r['Student ID']) === normId(p.studentId)); if (!row) return { error: 'الطالب غير موجود' }; const writingObj = parseJsonSafe(row['Writing']); writingObj[p.lecture] = p.writing || ''; const shadowingObj = parseJsonSafe(row['Shadowing']); shadowingObj[p.lecture] = p.shadowing || ''; setCell(sheet, headers, row.__row, 'Writing', JSON.stringify(writingObj)); setCell(sheet, headers, row.__row, 'Shadowing', JSON.stringify(shadowingObj)); return { success: true }; } /* ==================== حفظ نقاط اللعبة وجلب لوحة الصدارة ==================== */ function handleSubmitGameScore(p) { const { sheet, headers, rows } = getSheetData(SHEETS.STUDENTS); const row = rows.find(r => normId(r['Student ID']) === normId(p.studentId)); if (!row) return { error: 'الطالب غير موجود' }; const current = Number(row['Game points']) || 0; const added = Number(p.score) || 0; const newScore = current + added; setCell(sheet, headers, row.__row, 'Game points', newScore); row['Game points'] = newScore; const course = String(row['Course']).trim().toLowerCase(); const leaderboard = rows .filter(r => String(r['Course']).trim().toLowerCase() === course) .map(r => ({ name: String(r['Name'] || 'بدون اسم').trim(), score: Number(r['Game points']) || 0 })) .sort((a, b) => b.score - a.score); return { success: true, leaderboard: leaderboard }; } function handleSaveSchedule(p) { const sheet = SS.getSheetByName(SHEETS.SCHEDULE); const values = sheet.getDataRange().getValues(); let foundRow = -1; for (let i = 1; i < values.length; i++) { if (String(values[i][0]).trim() === String(p.group).trim() && String(values[i][1]).trim() === String(p.lecture).trim()) { foundRow = i + 1; break; } } const dateObj = new Date(p.dateTime); if (foundRow > 0) { sheet.getRange(foundRow, 3).setValue(dateObj); sheet.getRange(foundRow, 4).setValue(p.zoomLink || ''); } else { sheet.appendRow([p.group, p.lecture, dateObj, p.zoomLink || '']); } return { success: true }; } function handleUploadExam(p) { const sheet = SS.getSheetByName(SHEETS.EXAMS); const values = sheet.getDataRange().getValues(); let foundRow = -1; for (let i = 1; i < values.length; i++) { if (String(values[i][0]).trim().toLowerCase() === String(p.examKey).trim().toLowerCase()) { foundRow = i + 1; break; } } if (foundRow > 0) { sheet.getRange(foundRow, 2).setValue(p.examData); } else { sheet.appendRow([p.examKey, p.examData]); } return { success: true }; } function handleGradeHomework(p) { const { sheet, headers, rows } = getSheetData(SHEETS.STUDENTS); const row = rows.find(r => normId(r['Student ID']) === normId(p.studentId)); if (!row) return { error: 'الطالب غير موجود' }; const gradeObj = parseJsonSafe(row['HW_Grade']); gradeObj[p.lecture] = p.hwGrade || ''; const fbObj = parseJsonSafe(row['HW_Feedback']); fbObj[p.lecture] = p.hwFeedback || ''; setCell(sheet, headers, row.__row, 'HW_Grade', JSON.stringify(gradeObj)); setCell(sheet, headers, row.__row, 'HW_Feedback', JSON.stringify(fbObj)); return { success: true }; } function handleUpdateAttendance(p) { const { sheet, headers, rows } = getSheetData(SHEETS.ATTENDANCE); const row = rows.find(r => normId(r['ID']) === normId(p.studentId)); if (!row) { const { rows: studentRows } = getSheetData(SHEETS.STUDENTS); const student = studentRows.find(s => normId(s['Student ID']) === normId(p.studentId)); const name = student ? student['Name'] : ''; sheet.appendRow([p.studentId, name]); const newRowNum = sheet.getLastRow(); const colIdx = headers.indexOf(p.lecture) + 1; if (colIdx > 0) sheet.getRange(newRowNum, colIdx).setValue(p.status); return { success: true }; } setCell(sheet, headers, row.__row, p.lecture, p.status); return { success: true }; } function handleUpdateStudentFullInfo(p) { const sid = String(p.studentId).trim(); if (p.quiz1 !== undefined || p.quiz2 !== undefined || p.final !== undefined) { const { sheet, headers, rows } = getSheetData(SHEETS.GRADES); let row = rows.find(r => normId(r['ID']) === normId(sid)); if (!row) { sheet.appendRow([sid, '', '', '']); row = { __row: sheet.getLastRow() }; } if (p.quiz1 !== undefined) setCell(sheet, headers, row.__row, 'Quiz1', p.quiz1); if (p.quiz2 !== undefined) setCell(sheet, headers, row.__row, 'Quiz2', p.quiz2); if (p.final !== undefined) setCell(sheet, headers, row.__row, 'Final', p.final); } if (p.feedback !== undefined) { const { sheet, headers, rows } = getSheetData(SHEETS.STUDENTS); const srow = rows.find(r => normId(r['Student ID']) === normId(sid)); if (srow) setCell(sheet, headers, srow.__row, 'Feedback', p.feedback); } return { success: true }; }
